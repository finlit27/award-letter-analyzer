// Find the uploaded image binary and build the full Anthropic request body.
// IMPORTANT: on n8n Cloud, binary.data is a filesystem pointer (e.g. 'filesystem-v2'),
// NOT the actual base64. We must call getBinaryDataBuffer to read the real bytes.
const item = items[0];
const binaries = item.binary || {};
const keys = Object.keys(binaries);
if (keys.length === 0) {
  throw new Error('No binary file received. Available keys: ' + JSON.stringify(Object.keys(item)));
}
const key = binaries.pdfFile ? 'pdfFile' : keys[0];
const binary = binaries[key];
const buffer = await this.helpers.getBinaryDataBuffer(0, key);
const base64 = buffer.toString('base64');
const mime = binary.mimeType || 'image/jpeg';

const prompt = `You are a college financial aid analyst. Analyze this award letter image.

STEP 1 — IDENTIFY EACH LINE ITEM. Categorize every line by TYPE:

GIFT AID (free money — never repaid):
Keywords: Grant, Scholarship, Award, Merit, SEOG, Pell, TAP
- state_grant includes state-administered programs even when "state" is not in the name. Examples: Cal Grant A/B, CA College Promise Grant, NJ Tuition Aid Grant, Texas Grant, MA State Grant, Florida Bright Futures.
- institutional_merit = school-funded scholarships. Examples: school name in title (UC Grant, Pomona College Grant), Athletic Grant-in-Aid, Heritage Scholarship, Go Blue Guarantee Grant, Presidential/Achievement/Talent awards.

Keywords: Loan, Subsidized, Unsubsidized, PLUS, Direct, Perkins
- Always extract federal_unsubsidized if a Federal Direct Unsubsidized Loan is listed — do not drop it.

WORK-STUDY (earned money — student must work):
Keywords: Work-Study, Work Study, Employment, FWS, Federal Work-Study, Student Campus Job
- Always extract work_study if any work-study line appears — do not drop it.

COST OF ATTENDANCE buckets:
- direct_costs.tuition: "Tuition" or bundled "Tuition & Fees".
- direct_costs.housing: SUM of Room + Board + Meals + Dining + Food + Living Allowance. Always include board/meals — don't restrict to room only.

- Books, Personal Expenses, Transportation, Travel count toward total_cost_of_attendance but NOT direct_costs.

- total_gift_aid = grants + scholarships ONLY (no loans, no work-study). When the letter prints a "Gift Aid Subtotal" or "Total Gift Aid" line, read that number directly.

const makeBody = (model, text) => ({
  model,
  max_tokens: 2048,
  messages: [{
    role: 'user',
    content: [
      { type: 'image', source: { type: 'base64', media_type: mime, data: base64 } },
      { type: 'text', text }
    ]
  }]
});

return [{
  json: {
    haikuBody: makeBody('claude-haiku-4-5-20251001', prompt),
    sonnetBody: makeBody('claude-sonnet-4-6', strictPrompt),
    fileName: binary.fileName || 'unknown'
  }
}];
